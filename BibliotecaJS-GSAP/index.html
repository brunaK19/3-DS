GSAP - Animações

WaveMotion

gsap.to(".image img", { x: "+=100", rotation: 20, repeat: -1, yoyo: true, duration: 1, ease: "power1.inOut", stagger: 0.2 });

Fade

gsap.fromTo(".image img", { opacity: 0 }, { opacity: 1, repeat: -1, yoyo: true, duration: 1, ease: "power1.inOut" });

VerticalFade

gsap.fromTo(".image img", { opacity: 0, y: -50 }, { opacity: 1, y: 0, repeat: -1, yoyo: true, duration: 1, ease: "power1.inOut" });

HorizontalShift

gsap.fromTo(".image img", { x: -100 }, { x: 100, repeat: -1, yoyo: true, duration: 2, ease: "power2.inOut" });

Scale

gsap.fromTo(".image img", { scale: 0.5 }, { scale: 1.5, repeat: -1, yoyo: true, duration: 1, ease: "power1.inOut" });

Rotation

gsap.to(".image img", { rotation: 360, repeat: -1, duration: 2, ease: "power2.out" });
